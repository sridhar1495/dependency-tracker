TOKEN=$(curl -sf -X POST http://<DT-HOST>:8081/api/v1/user/login \ -d 'username=admin&password=YOUR_PASS') \ && curl -sf -H "Authorization: Bearer $TOKEN" \ http://<DT-HOST>:8081/api/v1/team | jq -r '.[0].apiKeys[0].key'
